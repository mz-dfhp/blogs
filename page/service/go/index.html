变量 ​

局部变量 ​

go
package main 
import "fmt"
func main() {
	fmt.Println("hello world")

	var age int = 18
	fmt.Println("age:", age)

	var age1 int
	fmt.Println("age1:", age1) // 0

	var age2 = 18
	fmt.Println("age2:", age2)

	name := "mz"
	fmt.Println("name:", name) 

	var n1, n2, n3 int
	fmt.Println(n1, n2, n3)

	var n4, n5, n6 int = 4, 5, 6
	fmt.Println(n4, n5, n6)

	n7, n8 := 7, 8
	fmt.Println(n7, n8)
}
